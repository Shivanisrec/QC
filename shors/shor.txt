import numpy as np

def M2mod15():
    qc = QuantumCircuit(4)
    qc.swap(2, 3)
    qc.swap(1, 2)
    qc.swap(0, 1)
    return qc.to_gate(label="M2")

def M4mod15():
    qc = QuantumCircuit(4)
    qc.swap(1, 3)
    qc.swap(0, 2)
    return qc.to_gate(label="M4")


def a2kmodN(a, k, N):
    return pow(a, 2**k, N)
