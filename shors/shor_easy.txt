import numpy as np
from qiskit import QuantumCircuit, QuantumRegister, ClassicalRegister, transpile
from qiskit.circuit.library import QFT
from qiskit_aer import AerSimulator
from qiskit.visualization import plot_histogram
import matplotlib.pyplot as plt


def mod_mult_gate(a_power):
    qc = QuantumCircuit(4)

    if a_power == 2:
        qc.swap(2, 3)
        qc.swap(1, 2)
        qc.swap(0, 1)

    elif a_power == 4:
        qc.swap(1, 3)
        qc.swap(0, 2)

    return qc.to_gate(label=f"M{a_power}")


def power_mod(a, k, N):
    return pow(a, 2**k, N)



def create_shor_circuit(a, N, num_control=8, num_target=4):

    control = QuantumRegister(num_control, "C")
    target = QuantumRegister(num_target, "T")
    classical = ClassicalRegister(num_control, "out")

    qc = QuantumCircuit(control, target, classical)

    qc.x(target[0])


    qc.h(control)

    for k in range(num_control):
        a_power = power_mod(a, k, N)

        if a_power in [2, 4]:
            gate = mod_mult_gate(a_power)
            qc.compose(gate.control(), [control[k]] + list(target), inplace=True)


    qc.append(QFT(num_control, inverse=True), control)


    qc.measure(control, classical)

    return qc



a = 2
N = 15

qc = create_shor_circuit(a, N)

sim = AerSimulator()
compiled = transpile(qc, sim)

result = sim.run(compiled, shots=1024).result()
counts = result.get_counts()

print(counts)

plot_histogram(counts)
plt.show()
