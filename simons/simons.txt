from qiskit import QuantumCircuit, QuantumRegister, ClassicalRegister
from qiskit_aer import AerSimulator
from qiskit.visualization import plot_histogram

n = 2
m = 2

oracle = QuantumCircuit(n + m)
oracle.cx(0, 2)
oracle.cx(1, 2)

qr_in = QuantumRegister(n)
qr_out = QuantumRegister(m)
cr = ClassicalRegister(n)
qc = QuantumCircuit(qr_in, qr_out, cr)

qc.h(qr_in)
qc.barrier()


qc.compose(oracle, inplace=True)
qc.barrier()

qc.h(qr_in)
qc.barrier()

qc.measure(qr_in, cr)

print("Drawing circuit diagram...")
display(qc.draw("mpl", fold=-1))

print("Running on local simulator...")


simulator = AerSimulator()
result = simulator.run(qc, shots=1024).result()


counts = result.get_counts()


display(plot_histogram(counts, title="Measurement Results"))

print("\n--- Classical Post-Processing ---")
print(f"Measured strings (y): {list(counts.keys())}")

possible_s_values = ['01', '10', '11']
secret_string = None

for s in possible_s_values:
    is_valid = True
    for y in counts.keys():
  
        dot_product = sum(int(y[i]) * int(s[i]) for i in range(n)) % 2

        if dot_product != 0:
            is_valid = False
            break

    if is_valid:
        secret_string = s
        break

if secret_string:
    print(f"*** Secret string successfully found: s = {secret_string} ***")
else:
    print("Could not determine the secret string (run more shots).")
